### 5.2 地图组件封装

```typescript
// apps/web/src/components/map/AmapContainer.tsx
'use client';

import { useEffect, useRef, useState } from 'react';
import AMapLoader from '@amap/amap-jsapi-loader';

interface AmapContainerProps {
  center?: [number, number];  // [lng, lat]
  zoom?: number;
  markers?: Array<{
    id: string;
    position: [number, number];
    title: string;
    status: 'idle' | 'busy' | 'full' | 'queue';
  }>;
  onMarkerClick?: (id: string) => void;
}

export function AmapContainer({
  center = [104.0668, 30.5728],  // 默认成都
  zoom = 15,
  markers = [],
  onMarkerClick,
}: AmapContainerProps) {
  const mapRef = useRef<HTMLDivElement>(null);
  const [map, setMap] = useState<any>(null);

  useEffect(() => {
    // 加载高德地图
    AMapLoader.load({
      key: process.env.NEXT_PUBLIC_AMAP_KEY!,  // 你的Key
      version: '2.0',
      plugins: ['AMap.Marker', 'AMap.InfoWindow', 'AMap.Scale'],
    })
      .then((AMap) => {
        // 初始化地图
        const mapInstance = new AMap.Map(mapRef.current!, {
          center,
          zoom,
          viewMode: '2D',
        });

        // 添加比例尺
        mapInstance.addControl(new AMap.Scale());

        setMap(mapInstance);
      })
      .catch((e) => {
        console.error('地图加载失败:', e);
      });

    return () => {
      map?.destroy();
    };
  }, []);

  // 更新标记点
  useEffect(() => {
    if (!map) return;

    // 清除旧标记
    map.clearMap();

    // 添加新标记
    markers.forEach((marker) => {
      const amapMarker = new window.AMap.Marker({
        position: marker.position,
        title: marker.title,
        // 自定义图标
        icon: new window.AMap.Icon({
          image: getMarkerIcon(marker.status),
          size: new window.AMap.Size(44, 44),
          imageSize: new window.AMap.Size(44, 44),
        }),
        offset: new window.AMap.Pixel(-22, -44),
      });

      amapMarker.on('click', () => {
        onMarkerClick?.(marker.id);
      });

      map.add(amapMarker);
    });
  }, [map, markers]);

  return (
    <div
      ref={mapRef}
      className="w-full h-full rounded-2xl overflow-hidden"
      style={{ minHeight: '500px' }}
    />
  );
}

// 根据状态获取标记图标
function getMarkerIcon(status: string): string {
  const icons = {
    idle: '/markers/marker-green.svg',
    busy: '/markers/marker-yellow.svg',
    full: '/markers/marker-red.svg',
    queue: '/markers/marker-purple.svg',
  };
  return icons[status as keyof typeof icons] || icons.idle;
}
```

### 5.4 热力图实现

```typescript
// apps/web/src/components/map/HeatmapLayer.tsx
'use client';

import { useEffect } from 'react';

interface HeatmapLayerProps {
  map: any;
  data: Array<{
    lng: number;
    lat: number;
    count: number;
  }>;
}

export function HeatmapLayer({ map, data }: HeatmapLayerProps) {
  useEffect(() => {
    if (!map || !data.length) return;

    // 加载热力图插件
    window.AMap.plugin(['AMap.HeatMap'], () => {
      const heatmap = new window.AMap.HeatMap(map, {
        radius: 25,
        opacity: [0, 0.8],
        // 自定义渐变色
        gradient: {
          0.25: '#E0F2FE',  // 冷
          0.5: '#FEF08A',   // 温
          0.75: '#FDBA74',  // 热
          1.0: '#EF4444',   // 爆
        },
      });

      heatmap.setDataSet({
        data: data.map((item) => ({
          lng: item.lng,
          lat: item.lat,
          count: item.count,
        })),
        max: 100,
      });
    });

    return () => {
      // 清理热力图
    };
  }, [map, data]);

  return null;
}
```

## 十、常见问题解决

### Q1: 高德地图显示"KEY错误"

```
解决:
1. 检查 Key 是否正确
2. 检查域名白名单是否包含当前域名
3. 检查是否启用了 JS API
4. 检查环境变量是否正确加载
```

### Q2: 地图标记不显示

```
解决:
1. 检查 markers 数据格式是否正确
2. 检查坐标系（高德使用 GCJ-02）
3. 检查图标路径是否正确
4. 查看浏览器控制台错误
```

### Q3: 热力图数据不更新

```
解决:
1. 检查后端数据聚合逻辑
2. 检查时间范围参数
3. 检查数据格式是否符合热力图要求
4. 检查最大值设置是否合理
```
