2. 详细表结构设计
2.1 用户模块 (User Module)
2.1.1 用户基础表 (users)
-- ============================================
-- 表名: users
-- 描述: 用户基础信息表
-- 数据量预估: 1万+ (单大学城)
-- ============================================

CREATE TABLE users (
    -- 主键
    id              BIGSERIAL PRIMARY KEY,
    
    -- 用户标识
    uuid            UUID NOT NULL DEFAULT gen_random_uuid(),
    openid          VARCHAR(64) UNIQUE,           -- 微信openid
    unionid         VARCHAR(64),                  -- 微信unionid
    
    -- 基础信息
    nickname        VARCHAR(50),                  -- 昵称
    avatar_url      VARCHAR(500),                 -- 头像URL
    phone           VARCHAR(20) UNIQUE,           -- 手机号
    email           VARCHAR(100),                 -- 邮箱
    
    -- 人口属性
    gender          SMALLINT DEFAULT 0,           -- 性别: 0未知 1男 2女
    birthday        DATE,                         -- 生日
    age_group       SMALLINT,                     -- 年龄段: 1-18-22 2-23-25 3-26+
    
    -- 学校信息
    university_id   BIGINT NOT NULL,              -- 所属大学ID
    campus_id       BIGINT,                       -- 校区ID
    enrollment_year SMALLINT,                     -- 入学年份
    
    -- 设备信息（用于分析和服务优化）
    device_type     VARCHAR(20),                  -- 设备类型: ios/android
    device_model    VARCHAR(50),                  -- 手机型号
    os_version      VARCHAR(30),                  -- 系统版本
    app_version     VARCHAR(20),                  -- App版本
    
    -- 位置信息（最近一次）
    last_location   GEOGRAPHY(POINT, 4326),       -- 最后已知位置
    last_loc_time   TIMESTAMP,                    -- 最后定位时间
    
    -- 账户状态
    status          SMALLINT DEFAULT 1,           -- 状态: 0禁用 1正常 2注销
    is_verified     BOOLEAN DEFAULT FALSE,        -- 是否认证（学生认证）
    verified_at     TIMESTAMP,                    -- 认证时间
    
    -- 统计字段（冗余设计，减少查询）
    review_count    INTEGER DEFAULT 0,            -- 评价数
    favorite_count  INTEGER DEFAULT 0,            -- 收藏数
    checkin_count   INTEGER DEFAULT 0,            -- 打卡数
    points          INTEGER DEFAULT 0,            -- 积分
    
    -- 时间戳
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW(),
    last_login_at   TIMESTAMP,                    -- 最后登录时间
    
    -- 软删除
    deleted_at      TIMESTAMP,
    is_deleted      BOOLEAN DEFAULT FALSE
);

-- 索引设计
CREATE INDEX idx_users_university ON users(university_id) WHERE is_deleted = FALSE;
CREATE INDEX idx_users_phone ON users(phone) WHERE is_deleted = FALSE;
CREATE INDEX idx_users_openid ON users(openid) WHERE is_deleted = FALSE;
CREATE INDEX idx_users_location ON users USING GIST(last_location) WHERE last_location IS NOT NULL;
CREATE INDEX idx_users_created_at ON users(created_at DESC);
2.1.2 用户偏好表 (user_preferences)
-- ============================================
-- 表名: user_preferences
-- 描述: 用户偏好设置（口味、预算、场景等）
-- ============================================

CREATE TABLE user_preferences (
    id              BIGSERIAL PRIMARY KEY,
    user_id         BIGINT NOT NULL UNIQUE,       -- 关联用户ID
    
    -- 口味偏好 (JSON数组存储)
    taste_tags      JSONB DEFAULT '[]',           -- ["辣", "甜", "清淡", "重口"]
    cuisine_prefs   JSONB DEFAULT '[]',           -- 偏好菜系 ["川菜", "粤菜"]
    avoid_tags      JSONB DEFAULT '[]',           -- 忌口 ["香菜", "花生"]
    
    -- 预算偏好
    price_range_min INTEGER DEFAULT 0,            -- 最低人均预算
    price_range_max INTEGER DEFAULT 100,          -- 最高人均预算
    
    -- 场景偏好
    scene_prefs     JSONB DEFAULT '[]',           -- ["快餐", "聚餐", "约会"]
    
    -- 距离偏好
    max_distance    INTEGER DEFAULT 1000,         -- 最大接受距离(米)
    
    -- 时间偏好
    prefer_dining_time JSONB DEFAULT '{}',        -- {"lunch": "11:30-13:00", ...}
    
    -- 推荐设置
    allow_push      BOOLEAN DEFAULT TRUE,         -- 允许推送
    push_time_start TIME DEFAULT '11:00',         -- 推送开始时间
    push_time_end   TIME DEFAULT '13:00',         -- 推送结束时间
    
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_user_prefs_user_id ON user_preferences(user_id);
2.1.3 用户行为表 (user_actions)
-- ============================================
-- 表名: user_actions
-- 描述: 用户行为日志（浏览、收藏、分享等）
-- 数据量预估: 日均10万+条
-- ============================================

CREATE TABLE user_actions (
    id              BIGSERIAL,
    user_id         BIGINT NOT NULL,
    
    -- 行为类型
    action_type     VARCHAR(30) NOT NULL,         -- view/browse/favorite/share/checkin/review/search
    
    -- 行为对象
    target_type     VARCHAR(20),                  -- merchant/product/region/page
    target_id       BIGINT,                       -- 对象ID
    
    -- 行为详情
    action_detail   JSONB DEFAULT '{}',           -- 详细参数
    -- 示例: {"merchant_id": 123, "from_page": "home", "stay_time": 30}
    
    -- 上下文信息
    context         JSONB DEFAULT '{}',           -- 上下文
    -- 示例: {"scene": "lunch", "weather": "sunny", "day_type": "weekday"}
    
    -- 位置信息
    location        GEOGRAPHY(POINT, 4326),       -- 行为发生位置
    
    -- 设备信息
    device_id       VARCHAR(64),                  -- 设备标识
    ip_address      INET,                         -- IP地址
    
    -- 时间戳
    action_time     TIMESTAMP NOT NULL DEFAULT NOW(),
    created_at      TIMESTAMP DEFAULT NOW()
) PARTITION BY RANGE (action_time);

-- 创建分区表示例
CREATE TABLE user_actions_202601 PARTITION OF user_actions
    FOR VALUES FROM ('2026-01-01') TO ('2026-02-01');

-- 索引
CREATE INDEX idx_user_actions_user_time ON user_actions(user_id, action_time DESC);
CREATE INDEX idx_user_actions_type_time ON user_actions(action_type, action_time DESC);
CREATE INDEX idx_user_actions_target ON user_actions(target_type, target_id);
2.1.4 用户收藏表 (user_favorites)
-- ============================================
-- 表名: user_favorites
-- 描述: 用户收藏关系
-- ============================================

CREATE TABLE user_favorites (
    id              BIGSERIAL PRIMARY KEY,
    user_id         BIGINT NOT NULL,
    target_type     VARCHAR(20) NOT NULL,         -- merchant/product
    target_id       BIGINT NOT NULL,
    
    -- 收藏时信息（快照）
    snapshot        JSONB DEFAULT '{}',           -- 收藏时的商家/商品信息
    
    -- 收藏夹分类
    folder_id       BIGINT DEFAULT 0,             -- 收藏夹ID (0为默认)
    
    -- 备注
    remark          VARCHAR(200),                 -- 用户备注
    
    created_at      TIMESTAMP DEFAULT NOW(),
    
    UNIQUE(user_id, target_type, target_id)
);

CREATE INDEX idx_favorites_user ON user_favorites(user_id, created_at DESC);
CREATE INDEX idx_favorites_target ON user_favorites(target_type, target_id);

2.2 商家模块 (Merchant Module)
2.2.1 商家基础表 (merchants)
-- ============================================
-- 表名: merchants
-- 描述: 商家基础信息表
-- 数据量预估: 500-1000家/大学城
-- ============================================

CREATE TABLE merchants (
    id                  BIGSERIAL PRIMARY KEY,
    
    -- 商家标识
    merchant_no         VARCHAR(20) UNIQUE NOT NULL,  -- 商家编号 M20260001
    
    -- 基本信息
    name                VARCHAR(100) NOT NULL,        -- 店铺名称
    short_name          VARCHAR(30),                  -- 简称
    logo_url            VARCHAR(500),                 -- Logo
    cover_images        JSONB DEFAULT '[]',           -- 封面图数组
    
    -- 从业资格信息
    business_license_no VARCHAR(50),                  -- 营业执照号
    business_license_img VARCHAR(500),                -- 营业执照照片
    food_license_no     VARCHAR(50),                  -- 食品经营许可证号
    food_license_img    VARCHAR(500),                 -- 食品经营许可证照片
    legal_person        VARCHAR(50),                  -- 法人姓名
    legal_person_idcard VARCHAR(20),                  -- 法人身份证
    
    -- 位置信息
    province            VARCHAR(20),                  -- 省
    city                VARCHAR(20),                  -- 市
    district            VARCHAR(20),                  -- 区
    address             VARCHAR(200) NOT NULL,        -- 详细地址
    location            GEOGRAPHY(POINT, 4326) NOT NULL, -- 经纬度坐标
    location_accuracy   INTEGER DEFAULT 10,           -- 定位精度(米)
    
    -- 关联区域
    region_id           BIGINT NOT NULL,              -- 所属区域ID（商场/街道等）
    university_id       BIGINT NOT NULL,              -- 主要服务大学
    
    -- 经营信息
    business_type       SMALLINT NOT NULL,            -- 经营类型: 1-餐饮 2-饮品 3-甜品 4-小吃 5-其他
    cuisine_types       JSONB DEFAULT '[]',           -- 菜系 ["川菜", "火锅"]
    main_dishes         JSONB DEFAULT '[]',           -- 主打菜品
    
    -- 店面规模
    store_size          SMALLINT,                     -- 规模: 1-小店(<50㎡) 2-中型(50-150㎡) 3-大型(>150㎡)
    seat_count          INTEGER DEFAULT 0,            -- 座位数
    has_private_room    BOOLEAN DEFAULT FALSE,        -- 是否有包厢
    private_room_count  INTEGER DEFAULT 0,            -- 包厢数量
    has_outdoor_seating BOOLEAN DEFAULT FALSE,        -- 是否有露天座位
    
    -- 服务类型
    has_dine_in         BOOLEAN DEFAULT TRUE,         -- 是否支持堂食
    has_takeaway        BOOLEAN DEFAULT FALSE,        -- 是否支持自取
    has_delivery        BOOLEAN DEFAULT FALSE,        -- 是否支持外卖
    delivery_platforms  JSONB DEFAULT '[]',           -- 外卖平台 ["meituan", "eleme"]
    
    -- 营业时间
    business_hours      JSONB DEFAULT '{}',           -- {"mon": {"open": "09:00", "close": "22:00"}}
    is_24h              BOOLEAN DEFAULT FALSE,        -- 是否24小时营业
    
    -- 商家定位
    price_per_person    INTEGER DEFAULT 0,            -- 人均消费(元)
    price_level         SMALLINT DEFAULT 1,           -- 价格档位: 1-便宜 2-适中 3-偏贵 4-贵
    serve_speed         SMALLINT DEFAULT 2,           -- 出餐速度: 1-快(<10min) 2-适中 3-慢(>20min)
    
    -- 场景标签
    scene_tags          JSONB DEFAULT '[]',           -- ["约会", "聚餐", "快餐", "深夜食堂"]
    feature_tags        JSONB DEFAULT '[]',           -- ["网红店", "老字号", "学生优惠"]
    
    -- 联系信息
    phone               VARCHAR(20),                  -- 联系电话
    wechat              VARCHAR(50),                  -- 微信号
    
    -- 社交账号
    social_accounts     JSONB DEFAULT '{}',           -- {"xiaohongshu": "xxx", "douyin": "xxx"}
    
    -- 状态管理
    status              SMALLINT DEFAULT 0,           -- 0-待审核 1-审核通过 2-审核拒绝 3-暂停营业 4-已关闭
    verify_status       SMALLINT DEFAULT 0,           -- 认证状态
    verify_time         TIMESTAMP,                    -- 认证时间
    
    -- 实时状态（商家主动更新或系统自动检测）
    realtime_status     SMALLINT DEFAULT 0,           -- 0-未知 1-空闲 2-忙碌 3-满客 4-排队 5-休息中
    queue_count         INTEGER DEFAULT 0,            -- 当前排队人数
    wait_time_estimate  INTEGER DEFAULT 0,            -- 预计等待时间(分钟)
    status_updated_at   TIMESTAMP,                    -- 状态更新时间
    
    -- 统计字段（冗余，定时更新）
    view_count          INTEGER DEFAULT 0,            -- 浏览次数
    favorite_count      INTEGER DEFAULT 0,            -- 收藏数
    checkin_count       INTEGER DEFAULT 0,            -- 打卡数
    review_count        INTEGER DEFAULT 0,            -- 评价数
    review_avg_score    DECIMAL(2,1) DEFAULT 5.0,     -- 平均评分
    
    -- 评分细分
    score_taste         DECIMAL(2,1) DEFAULT 5.0,     -- 口味评分
    score_environment   DECIMAL(2,1) DEFAULT 5.0,     -- 环境评分
    score_service       DECIMAL(2,1) DEFAULT 5.0,     -- 服务评分
    score_price         DECIMAL(2,1) DEFAULT 5.0,     -- 性价比评分
    
    -- 销量统计
    sales_monthly       INTEGER DEFAULT 0,            -- 月销量
    sales_total         INTEGER DEFAULT 0,            -- 总销量
    
    -- 时间戳
    created_at          TIMESTAMP DEFAULT NOW(),
    updated_at          TIMESTAMP DEFAULT NOW(),
    
    -- 软删除
    deleted_at          TIMESTAMP,
    is_deleted          BOOLEAN DEFAULT FALSE
);

-- 索引设计
CREATE INDEX idx_merchants_status ON merchants(status) WHERE is_deleted = FALSE;
CREATE INDEX idx_merchants_location ON merchants USING GIST(location) WHERE is_deleted = FALSE;
CREATE INDEX idx_merchants_university ON merchants(university_id, status) WHERE is_deleted = FALSE;
CREATE INDEX idx_merchants_region ON merchants(region_id) WHERE is_deleted = FALSE;
CREATE INDEX idx_merchants_type ON merchants(business_type, status) WHERE is_deleted = FALSE;
CREATE INDEX idx_merchants_price ON merchants(price_per_person) WHERE is_deleted = FALSE;
CREATE INDEX idx_merchants_score ON merchants(review_avg_score DESC) WHERE is_deleted = FALSE;
CREATE INDEX idx_merchants_realtime ON merchants(realtime_status) WHERE is_deleted = FALSE;
2.2.2 商家动态状态表 (merchant_status_logs)
-- ============================================
-- 表名: merchant_status_logs
-- 描述: 商家状态变更日志（用于分析客流规律）

    -- 时间戳
    recorded_at     TIMESTAMP NOT NULL DEFAULT NOW(),
    created_at      TIMESTAMP DEFAULT NOW()
) PARTITION BY RANGE (recorded_at);

2.3 商品模块 (Product Module)
2.3.1 商品基础表 (products)
-- ============================================
-- 表名: products
-- 描述: 商品/菜品信息表
-- 数据量预估: 5000-10000个/大学城
-- ============================================

CREATE TABLE products (
    id              BIGSERIAL PRIMARY KEY,
    
    -- 商品标识
    product_no      VARCHAR(20) UNIQUE NOT NULL,  -- 商品编号 P20260001
    
    -- 关联信息
    merchant_id     BIGINT NOT NULL,              -- 所属商家ID
    
    -- 基本信息
    name            VARCHAR(100) NOT NULL,        -- 商品名称
    sub_title       VARCHAR(200),                 -- 副标题/简介
    description     TEXT,                         -- 详细描述
    
    -- 图片
    main_image      VARCHAR(500),                 -- 主图
    images          JSONB DEFAULT '[]',           -- 图片数组
    
    -- 分类
    category_id     BIGINT NOT NULL,              -- 所属分类ID
    category_path   VARCHAR(100),                 -- 分类路径 1/2/3
    
    -- 商品类型
    product_type    SMALLINT DEFAULT 1,           -- 1-单品 2-套餐 3-规格商品
    
    -- 价格信息
    original_price  DECIMAL(10,2) DEFAULT 0,      -- 原价
    sale_price      DECIMAL(10,2) DEFAULT 0,      -- 售价
    member_price    DECIMAL(10,2),                -- 会员价
    
    -- 规格信息（针对有规格的商品）
    specifications  JSONB DEFAULT '{}',           -- {"size": ["大", "中", "小"], "temperature": ["冰", "热"]}
    
    -- 属性标签
    taste_tags      JSONB DEFAULT '[]',           -- 口味标签 ["辣", "麻"]
    ingredient_tags JSONB DEFAULT '[]',           -- 食材标签 ["牛肉", "土豆"]
    dietary_tags    JSONB DEFAULT '[]',           -- 饮食标签 ["素食", "低脂"]
    
    -- 商品属性
    is_recommended  BOOLEAN DEFAULT FALSE,        -- 是否推荐
    is_signature    BOOLEAN DEFAULT FALSE,        -- 是否招牌菜
    is_new          BOOLEAN DEFAULT FALSE,        -- 是否新品
    is_spicy        BOOLEAN DEFAULT FALSE,        -- 是否辣
    
    -- 库存/供应
    is_available    BOOLEAN DEFAULT TRUE,         -- 是否可售
    availability_note VARCHAR(100),               -- 不可售说明
    
    -- 统计字段
    view_count      INTEGER DEFAULT 0,            -- 浏览次数
    order_count_7d  INTEGER DEFAULT 0,            -- 7天销量
    order_count_30d INTEGER DEFAULT 0,            -- 30天销量
    order_count_90d INTEGER DEFAULT 0,            -- 90天销量
    order_count_total INTEGER DEFAULT 0,          -- 总销量
    
    -- 评价统计
    review_count    INTEGER DEFAULT 0,            -- 评价数
    review_avg_score DECIMAL(2,1) DEFAULT 5.0,    -- 平均评分
    review_good_count INTEGER DEFAULT 0,          -- 好评数
    review_mid_count INTEGER DEFAULT 0,           -- 中评数
    review_bad_count INTEGER DEFAULT 0,           -- 差评数
    
    -- 排序权重
    sort_order      INTEGER DEFAULT 0,            -- 手动排序
    
    -- 状态
    status          SMALLINT DEFAULT 1,           -- 0-下架 1-上架 2-售罄
    
    -- 时间戳
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW(),
    
    -- 软删除
    deleted_at      TIMESTAMP,
    is_deleted      BOOLEAN DEFAULT FALSE
);

CREATE INDEX idx_products_merchant ON products(merchant_id, status) WHERE is_deleted = FALSE;
CREATE INDEX idx_products_category ON products(category_id) WHERE is_deleted = FALSE;
CREATE INDEX idx_products_status ON products(status) WHERE is_deleted = FALSE;
CREATE INDEX idx_products_sales ON products(order_count_30d DESC) WHERE is_deleted = FALSE;
CREATE INDEX idx_products_score ON products(review_avg_score DESC) WHERE is_deleted = FALSE;

2.4 地区模块 (Region Module)
2.4.1 地区基础表 (regions)
-- ============================================
-- 表名: regions
-- 描述: 地理区域表（商场、街道、底商聚集区等）
-- ============================================

CREATE TABLE regions (
    id              BIGSERIAL PRIMARY KEY,
    
    -- 区域标识
    region_no       VARCHAR(20) UNIQUE NOT NULL,  -- 区域编号 R20260001
    
    -- 基本信息
    name            VARCHAR(100) NOT NULL,        -- 区域名称
    short_name      VARCHAR(30),                  -- 简称
    
    -- 区域类型
    region_type     SMALLINT NOT NULL,            -- 1-底商聚集区 2-商场 3-美食街 4-农家乐 5-校园内 6-其他
    
    -- 位置信息
    province        VARCHAR(20),
    city            VARCHAR(20),
    district        VARCHAR(20),
    address         VARCHAR(200),                 -- 大致地址
    
    -- 地理边界（多边形）
    boundary        GEOGRAPHY(POLYGON, 4326),     -- 区域边界
    center_point    GEOGRAPHY(POINT, 4326),       -- 中心点
    
    -- 关联大学
    university_id   BIGINT NOT NULL,              -- 主要服务大学
    distance_to_uni INTEGER,                      -- 距离大学(米)
    
    -- 区域属性
    has_parking     BOOLEAN DEFAULT FALSE,        -- 是否有停车场
    has_wifi        BOOLEAN DEFAULT FALSE,        -- 是否有公共WiFi
    is_pedestrian   BOOLEAN DEFAULT TRUE,         -- 是否步行街

-- ============================================
-- 表名: region_heat_stats
-- 描述: 地区热力统计数据（用于热力图展示）

    region_id       BIGINT NOT NULL,

    customer_count  INTEGER DEFAULT 0,            -- 该小时客流量

-- ============================================
-- 表名: reviews

    target_type     VARCHAR(20) NOT NULL DEFAULT 'merchant',  -- merchant/product

    updated_at      TIMESTAMP DEFAULT NOW(),
    
    -- 软删除
    deleted_at      TIMESTAMP,
    is_deleted      BOOLEAN DEFAULT FALSE
);
